# 001
chuu 'uuqhhli'aqtls qwis\hhitiich quu'as hiniip 'ink.

# 002
wik'iitukwitwa'ish 'ink quu'as huu'ak 'uyi.

# 003
wik'iit qwiihhw'al'aahhii'al siqiil.

# 004
'ana 'unaak 'ink qwayac'iikshtaquml.

# 005
tuhh'in\'al qwayac'iikshtaquml.

# 006
'ayup'it quu'as n'amilshitl hiinip 'ink\'ii.

# 007
wim'aaqtl'atlquuch\'al tlawiichitl.

# 008
huu'asshiil, hu'in\'atlquuchtla, wim'aaqtl'at\'ish waa.

# 009
chaa waa katlhhshitl hisumyis'iswa'ish hhaa muwaachm'it 'umch'aap.

# 010
na'aa 'an ciiqciiqa hhaa 'uu'asm'inhh'i chaakupiihh 'anch wim'aaqtl'at tlawiichi'at hiy'at\'itq qwayac'iikshtaquml.

# 011
yacins'ii'atl quchknahhiy'ul'is'i muwaachm'it 'u'umhhis hiinipchip siihhil.

# 012
tl'iixc'uy'inchi'atl muwaachm'it.

# 013
'aqis\hhcumch 'u''asht waa'atl'at, wim'aaqtlukna haa 'a'iihhp'itukqin hiinip 'ink\'ii, tl'iixc'uy'in\'atl.

# 014
hu'aas'atltla waa muwaachm'it hiinipchip\'aqtl siihhil 'ink waa.

# 015
tl'iiw'aaqichiilshi'atltla.

# 016
haa n'amilsap'i waa'atl hhaw'il'i n'amilsap'i.

# 017
chuu waa'atl'at.

# 018
'ee nuchhhakshitl\'ic'atl xhaxh chaaxtshitl muwaachm'it.

# 019
chuu 'un'aahhchip'is qwikw'ichhh'aahhiis huulhuula.
